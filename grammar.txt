STATEMENT: Int/Float identifier = EXPRESSION;
         : Int/Float identifier;
         : EXPRESSION;

EXPRESSION: identifier = COMP_EXPRESSION ( && | || COMP_EXPRESSION)*

COMP_EXPRESSION: ! COMP_EXPRESSION
               : ARITH_EXPRESSION ((== | != | < | > | <= | >=) ARITH_EXPRESSION)*

ARITH_EXPRESSION : TERM (PLUS | MINUS TERM)*

TERM: FACTOR (MULT | DIV FACTOR)*

FACTOR: (+*-* EXPRESSION)*
      : POWER

POWER: ATOM (^ FACTOR)*

ATOM: identifier
    : INT, FLOAT
    : (EXPRESSION)
    : Cast<Int/Float>(EXPRESSION)
